# ================================================================
# test_passwords.txt — 25 test cases cho password strength checker
# Dùng với: password_utils.py
#   - meets_password_requirements(): True/False
#   - get_password_strength(): Weak / Medium / Strong
#
# Yêu cầu hệ thống:
#   - Tối thiểu 6 ký tự
#   - Phải có: chữ HOA, chữ thường, số, ký tự đặc biệt
#   - Weak   = không đủ yêu cầu trên
#   - Medium = đủ yêu cầu, độ dài 6–8 ký tự
#   - Strong = đủ yêu cầu, độ dài > 8 ký tự
# ================================================================

# ----------------------------------------------------------------
# NHÓM 1: WEAK — Quá ngắn (< 6 ký tự)
# ----------------------------------------------------------------
abc
12345
A1!

# ----------------------------------------------------------------
# NHÓM 2: WEAK — Đủ độ dài nhưng thiếu loại ký tự
# ----------------------------------------------------------------
# Chỉ có chữ thường + số (thiếu hoa + đặc biệt)
password123
abcdef99

# Chỉ có chữ hoa + số (thiếu thường + đặc biệt)
PASSWORD123
SECURE99AB

# Đủ 3 loại nhưng thiếu ký tự đặc biệt
MyPassword1
Secure2026A

# Đủ 3 loại nhưng thiếu chữ hoa
mypassword1!
secure@2026

# Đủ 3 loại nhưng thiếu số
MyPassword!
Secure@Pass

# ----------------------------------------------------------------
# NHÓM 3: MEDIUM — Đủ yêu cầu, độ dài đúng 6–8 ký tự
# ----------------------------------------------------------------
Ab1!abc

Mn2$qrst
